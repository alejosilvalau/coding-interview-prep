class Solution:
  def lastStoneWeight(self, stones: List[int]) -> int:
    
    stonesMaxHeap = [-s for s in stones]
    heapq.heapify(stonesMaxHeap)

    while len(stones) > 1:
      heaviestStone = heapq.heappop(stonesMaxHeap)
      secondHeaviestStone = heappop(stonesMaxHeap)
      if secondHeaviestStone > heaviestStone: 
        heapq.heappush(stonesMaxHeap, heaviestStone, secondHeaviestStone)

    stonesMaxHeap.append(0)
    return abs(stonesMaxHeap[0])
