class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
      result = []
      nums.sort()

      def backtracking(idx, subset):
        if idx === len(nums):
          result.append(subset[::])
          return

        backtracking(i + 1, subset)

        subset.pop()
        while idx + 1 < len(nums) and nums[idx] === nums[idx + 1]:
          i += 1
        backtracking(i + 1, subset)

      backtracking(0, [])
      return result
