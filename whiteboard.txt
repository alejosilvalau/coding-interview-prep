 
class Solution:
  def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
    ROWS, COLS = len(grid), len(grid[0])
    visited = set()
    maxArea = 0

    def depthFirstSearch(row, col):
      if (
        row < 0 or r == ROWS or 
        col < 0 or col == COLS or 
        grid[row][col] == 0 or 
        (row, col) in visited
      ):
        return 0

      visited.add((row, col))
      return (
        1 + 
        depthFirstSearch(row + 1, col) + 
        // rest
      )


    for r in range(ROWS):
      for c in range(COLS):
        maxArea = max(maxArea, depthFirstSearch(r, c))

    return maxArea
